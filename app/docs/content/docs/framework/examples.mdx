# Examples

## Overview

This section provides comprehensive examples of how to use Axiomkit for various real-world scenarios. Each example includes complete implementation details and best practices.

## 🤖 Basic Examples

### Simple Chat Agent

```typescript
import { createAgent } from "@axiomkit/core";
import { groq } from "@ai-sdk/groq";

const agent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  modelSettings: {
    maxTokens: 1000,
    temperature: 0.7,
  },
});

async function main() {
  await agent.start();
  
  const result = await agent.run({
    context: {
      type: "chat",
      instructions: "You are a helpful assistant. Be concise and friendly.",
    },
    args: { message: "Hello! How are you today?" },
  });
  
  console.log("Agent:", result.output);
  await agent.stop();
}

main().catch(console.error);
```

### Calculator Bot

```typescript
import { createAgent, createAction } from "@axiomkit/core";
import { groq } from "@ai-sdk/groq";

const calculateAction = createAction({
  name: "calculate",
  description: "Perform mathematical calculations",
  parameters: {
    expression: { type: "string", description: "Mathematical expression" },
    precision: { type: "number", default: 2, description: "Decimal precision" }
  },
  execute: async ({ expression, precision }) => {
    try {
      const result = eval(expression);
      return {
        result: Number(result.toFixed(precision)),
        expression,
        success: true
      };
    } catch (error) {
      return {
        error: "Invalid expression",
        expression,
        success: false
      };
    }
  }
});

const calculatorAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  actions: [calculateAction],
  context: {
    type: "calculator",
    instructions: [
      "You are a helpful calculator assistant",
      "Use the calculate action for mathematical operations",
      "Always show your work and explain results"
    ]
  }
});
```

## 📱 Platform Integrations

### Telegram Customer Support Bot

```typescript
import { createAgent } from "@axiomkit/core";
import { createTelegramExtension } from "@axiomkit/telegram";
import { createMongoDBExtension } from "@axiomkit/mongodb";
import { groq } from "@ai-sdk/groq";

const createTicket = createAction({
  name: "create_ticket",
  description: "Create a support ticket",
  parameters: {
    customerId: { type: "string", description: "Customer identifier" },
    issue: { type: "string", description: "Issue description" },
    priority: { type: "string", enum: ["low", "medium", "high"], default: "medium" }
  },
  execute: async ({ customerId, issue, priority }) => {
    const ticket = await supportSystem.createTicket({
      customerId, issue, priority, createdAt: new Date()
    });
    return { ticketId: ticket.id, status: "created" };
  }
});

const telegramExt = createTelegramExtension({
  botToken: process.env.TELEGRAM_BOT_TOKEN,
  commands: {
    "/start": "Welcome to customer support!",
    "/help": "Available commands: /start, /help, /ticket",
    "/ticket": "Describe your issue to create a ticket."
  },
  actions: [createTicket]
});

const mongoExt = createMongoDBExtension({
  uri: process.env.MONGODB_URI,
  database: "customer_support",
  collections: {
    conversations: "conversations",
    tickets: "tickets",
    customers: "customers"
  }
});

const supportAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  extensions: [telegramExt, mongoExt],
  context: {
    type: "customer_support",
    instructions: [
      "You are a professional customer support agent",
      "Be helpful, patient, and professional",
      "Create tickets for issues requiring human intervention",
      "Remember customer history and preferences"
    ]
  }
});

supportAgent.start().then(() => {
  console.log("Customer support bot is running!");
});
```

### Discord Gaming Bot

```typescript
import { createAgent } from "@axiomkit/core";
import { createDiscordExtension } from "@axiomkit/discord";
import { groq } from "@ai-sdk/groq";

const rollDice = createAction({
  name: "roll_dice",
  description: "Roll dice for tabletop games",
  parameters: {
    dice: { type: "string", description: "Dice notation (e.g., '2d6', '1d20')" }
  },
  execute: async ({ dice }) => {
    const [count, sides] = dice.split('d').map(Number);
    const rolls = Array.from({ length: count }, () => 
      Math.floor(Math.random() * sides) + 1
    );
    return {
      dice,
      rolls,
      total: rolls.reduce((sum, roll) => sum + roll, 0)
    };
  }
});

const discordExt = createDiscordExtension({
  token: process.env.DISCORD_BOT_TOKEN,
  intents: ["GuildMessages", "DirectMessages"],
  commands: {
    "!roll": "Roll dice (e.g., !roll 2d6)",
    "!help": "Show available commands"
  },
  actions: [rollDice]
});

const gamingAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  extensions: [discordExt],
  context: {
    type: "gaming",
    instructions: [
      "You are a helpful gaming assistant",
      "Help with dice rolls, character generation, and game rules",
      "Be enthusiastic and encouraging"
    ]
  }
});
```

### E-commerce Assistant

```typescript
import { createAgent } from "@axiomkit/core";
import { createMongoDBExtension } from "@axiomkit/mongodb";
import { groq } from "@ai-sdk/groq";

const searchProducts = createAction({
  name: "search_products",
  description: "Search for products",
  parameters: {
    query: { type: "string", description: "Search query" },
    category: { type: "string", optional: true, description: "Product category" },
    limit: { type: "number", default: 10, description: "Maximum results" }
  },
  execute: async ({ query, category, limit }) => {
    const products = await productService.search(query, { category, limit });
    return {
      products: products.map(p => ({
        id: p.id,
        name: p.name,
        price: p.price,
        category: p.category,
        inStock: p.inStock
      })),
      total: products.length
    };
  }
});

const mongoExt = createMongoDBExtension({
  uri: process.env.MONGODB_URI,
  database: "ecommerce",
  collections: {
    products: "products",
    orders: "orders",
    customers: "customers"
  }
});

const ecommerceAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  extensions: [mongoExt],
  actions: [searchProducts],
  context: {
    type: "ecommerce",
    instructions: [
      "You are a helpful e-commerce assistant",
      "Help customers find products and place orders",
      "Provide accurate pricing and availability information"
    ]
  }
});
```

### Multi-Agent System

```typescript
import { createAgent } from "@axiomkit/core";
import { groq } from "@ai-sdk/groq";

// Research Agent
const researchAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  context: {
    type: "research",
    instructions: [
      "You are a research specialist",
      "Gather and analyze information from multiple sources",
      "Provide comprehensive reports with citations"
    ]
  }
});

// Writing Agent
const writingAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  context: {
    type: "writing",
    instructions: [
      "You are a professional writer",
      "Create engaging, well-structured content",
      "Adapt writing style to target audience"
    ]
  }
});

// Example workflow
async function researchAndWrite(topic: string) {
  const research = await researchAgent.run({
    args: { message: `Research topic: ${topic}` }
  });
  
  const writing = await writingAgent.run({
    args: { 
      message: `Write an article about ${topic} using this research: ${research.output}` 
    }
  });
  
  return writing.output;
}
```

### Memory-Enhanced Assistant

```typescript
import { createAgent } from "@axiomkit/core";
import { createMongoDBExtension } from "@axiomkit/mongodb";
import { groq } from "@ai-sdk/groq";

const mongoExt = createMongoDBExtension({
  uri: process.env.MONGODB_URI,
  database: "assistant_memory",
  collections: {
    conversations: "conversations",
    knowledge: "knowledge",
    preferences: "preferences"
  }
});

const memoryAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  extensions: [mongoExt],
  context: {
    type: "memory_enhanced",
    instructions: [
      "You are a memory-enhanced assistant",
      "Remember user preferences and past interactions",
      "Use context from previous conversations",
      "Learn from user feedback and adapt accordingly"
    ]
  }
});

async function chatWithMemory(userId: string, message: string) {
  const history = await memoryAgent.memory.query({
    question: `Previous conversations with user ${userId}`,
    limit: 5
  });
  
  const result = await memoryAgent.run({
    context: { type: "chat", userId, history },
    args: { message }
  });
  
  await memoryAgent.memory.store({
    content: `User ${userId}: ${message}\nAssistant: ${result.output}`,
    metadata: { userId, timestamp: new Date(), type: "conversation" }
  });
  
  return result.output;
}
```

## 🚀 Production Examples

### API Gateway Agent

```typescript
import { createAgent } from "@axiomkit/core";
import { createMongoDBExtension } from "@axiomkit/mongodb";
import { groq } from "@ai-sdk/groq";
import express from "express";

const app = express();
app.use(express.json());

const mongoExt = createMongoDBExtension({
  uri: process.env.MONGODB_URI,
  database: "api_gateway"
});

const apiAgent = createAgent({
  model: groq("deepseek-r1-distill-llama-70b"),
  extensions: [mongoExt],
  context: {
    type: "api_gateway",
    instructions: [
      "You are an API gateway assistant",
      "Route requests to appropriate services",
      "Validate and sanitize inputs",
      "Provide helpful error messages"
    ]
  }
});

app.post("/api/chat", async (req, res) => {
  try {
    const { message, userId, context } = req.body;
    
    const result = await apiAgent.run({
      context: { type: "api", userId, ...context },
      args: { message }
    });
    
    res.json({
      success: true,
      response: result.output,
      timestamp: new Date().toISOString()
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

app.listen(3000, () => {
  console.log("API Gateway running on port 3000");
});
```

## 📚 Getting Started

### 1. Choose Your Example

Start with a basic example that matches your use case:
- **Simple Chat** - For basic conversation agents
- **Calculator Bot** - For mathematical operations
- **Telegram Bot** - For messaging platforms
- **MongoDB Integration** - For data persistence
- **Sei Integration** - For blockchain applications

### 2. Set Up Dependencies

```bash
# Core framework
npm install @axiomkit/core

# Extensions (choose based on your example)
npm install @axiomkit/telegram @axiomkit/discord @axiomkit/mongodb @axiomkit/sei

# LLM providers
npm install @ai-sdk/groq @ai-sdk/openai @ai-sdk/anthropic
```

### 3. Configure Environment

```bash
# LLM API keys
GROQ_API_KEY=your_groq_key
OPENAI_API_KEY=your_openai_key

# Extension configurations
TELEGRAM_BOT_TOKEN=your_telegram_token
MONGODB_URI=your_mongodb_uri
SEI_RPC_URL=your_sei_rpc_url
```

### 4. Run the Example

```bash
# Compile TypeScript
npx tsc

# Run the example
node dist/your-example.js
```

## Best Practices

### 1. Error Handling

```typescript
try {
  const result = await agent.run({
    context: contextConfig,
    args: userInput
  });
  return result.output;
} catch (error) {
  console.error("Agent error:", error);
  return "I encountered an error. Please try again.";
}
```

### 2. Rate Limiting

```typescript
import rateLimit from "express-rate-limit";

const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutes
  max: 100 // limit each IP to 100 requests per windowMs
});

app.use("/api/", limiter);
```

### 3. Logging

```typescript
import winston from "winston";

const logger = winston.createLogger({
  level: "info",
  format: winston.format.json(),
  transports: [
    new winston.transports.File({ filename: "error.log", level: "error" }),
    new winston.transports.File({ filename: "combined.log" })
  ]
});

const result = await agent.run(config);
logger.info("Agent interaction", {
  input: config.args,
  output: result.output,
  timestamp: new Date()
});
```
